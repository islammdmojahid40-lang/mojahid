        // --- PRAYER TIMES ---
        async function fetchPrayers() {
            const res = await fetch('https://api.aladhan.com/v1/timingsByCity?city=Dhaka&country=Bangladesh&method=2');
            const data = await res.json();
            const t = data.data.timings;
            document.getElementById('sehri-time').innerText = t.Imsak;
            document.getElementById('iftar-time').innerText = t.Maghrib;
            document.getElementById('hijri-date').innerText = data.data.date.hijri.day;
            
            const list = [{n:'ржлржЬрж░', t:t.Fajr}, {n:'ржпрзЛрж╣рж░', t:t.Dhuhr}, {n:'ржЖрж╕рж░', t:t.Asr}, {n:'ржорж╛ржЧрж░рж┐ржм', t:t.Maghrib}, {n:'ржПрж╢рж╛', t:t.Isha}];
            document.getElementById('prayer-grid').innerHTML = list.map(p => `
                <div class="bg-white dark:bg-slate-900 p-4 rounded-2xl border dark:border-slate-800 text-center">
                    <p class="text-[10px] font-bold text-slate-400 uppercase">${p.n}</p>
                    <p class="font-black text-emerald-800 dark:text-emerald-400">${p.t}</p>
                </div>
            `).join('');
        }

        // --- TRACKER ---
        function updateTracker() {
            const keys = Object.keys(prayers);
            const done = keys.filter(k => prayers[k]).length;
            const pct = Math.round((done/5)*100);
            document.getElementById('tracker-pct').innerText = pct + '%';
            document.getElementById('tracker-btns').innerHTML = keys.map(k => `
                <button onclick="toggleP('${k}')" class="py-2 rounded-xl text-[10px] font-bold border transition-all ${prayers[k]?'bg-amber-400 border-amber-400 text-emerald-900':'bg-slate-100 dark:bg-slate-800 border-transparent text-slate-400'}">${k}</button>
            `).join('');
            localStorage.setItem('mojahid_p', JSON.stringify(prayers));
        }
        function toggleP(k) { prayers[k] = prayers[k]?0:1; updateTracker(); }
